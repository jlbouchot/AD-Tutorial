\section{Real valued functions}

%================================================
\begin{frame}{Three Approaches (1/3)}

\begin{block}{Finite Differences}
\begin{itemize}
\item Easy
\item Numerical error
\item Expensive in high dimension
\item Parametrization
\end{itemize}
\end{block}

\begin{block}{Example}
Let $f: \Rbb^p \to \Rbb$. Each derivative implies the computation of 
\begin{equation*}
\frac{\partial f}{\partial \xbf_i} \approx \frac{f(\xbf + \varepsilon \ebf_i) - f(\xbf [- \varepsilon \ebf_i])}{[2]\varepsilon}
\end{equation*}	
Computing the gradient costs $[2 \times] p$ the cost of a single function evaluation.
\end{block}

\end{frame}

\begin{frame}{Three Approaches (2/3)}
\begin{block}{Symbolic Differentiation}
\begin{itemize}
\item Exact
\item Expression swell
\item Hard for real codes
\end{itemize}
\end{block}

\begin{block}{Example}
Consider the function 
$f: \Rbb \to \Rbb$ defined as $f(x) = x^2\sin(x^2/2 + 3x)$. 

Its symbolic derivative is given by 
\[
\frac{\dif{f}}{\dif x} (x) = 2x\sin(x^2/2+3x) + x^2(x+3)\cos(x^2/2+3x)
\]
\end{block}

\end{frame}

\begin{frame}{Three Approaches (3/3)}
\begin{block}{Algorithmic Differentiation}
\begin{itemize}
\item Exact up to machine precision
\item Works on programs
\item Efficient
\end{itemize}
\end{block}

\begin{block}

\begin{itemize}
\item AD differentiates code, not formula!
\item AD factorizes / reuses variables
\end{itemize}

\end{block}

\end{frame}

\section{Forward Mode}

%================================================
\begin{frame}{Forward diff by dual numbers}

\begin{block}{Dual numbers}
Let every number embed its differentiated value. 
Define:

\[
\xbf = x + \dot{x}\varepsilon
\]
with the fictitious number $\varepsilon^2 = 0$.
\end{block}

$\varepsilon$ is not \emph{some small number}, but redefines the arithmetic on real numbers.

\end{frame}

\begin{frame}{Dual number propagation}

We propagate derivatives together with values.

\[
\xbf \rightarrow (x, \dot{x})
\]

Each operation updates both.

\pause

\begin{block}{Example}
Consider the univariate function $f(x) = x^2$.
\[
\xbf^2 = (x + \dot{x}\varepsilon)^2
= x^2 + 2x\dot{x}\varepsilon \left[ + \dot{x}^2 \varepsilon^2 \right].
\]

\end{block}

\begin{block}{Key insight}
Forward AD is ordinary arithmetic on dual numbers.
\end{block}

\end{frame}

%================================================
\begin{frame}{Forward Mode Example}

Given the function $z = f(x) = \sin(x^2)$

Propagation table:

\begin{center}
\begin{tabular}{c|c|c}
Variable & Value & Derivative \\
\hline
$x$ & $x$ & $\dot{x}$ \\
$y=x^2$ & $x^2$ & $2x\dot{x}$ \\
$z=\sin(y)$ & $\sin(y)$ & $\cos(y)\cdot 2x\dot{x}$
\end{tabular}
\end{center}

\begin{block}{Let's try}
\begin{itemize}
\item $f(x) = \sin(x)$
\item $f(x) = \cos(x)$
\item $f(x) = \sqrt{x}$
\end{itemize}
\end{block}

\end{frame}

\begin{frame}{Let's try}
  \begin{block}{$\sin$}
    \vspace{-0.5cm}
    \begin{align*}
      \sin(\xbf) &= \sin(x)\cos(\dot{x}\varepsilon) + \cos(x)\sin(\dot{x}\varepsilon) \\
      &= \sin(x)\left( 1 + \mathcal{O}(\dot{x}^2\varepsilon^2) \right) + \cos(x)(\dot{x}\varepsilon + \mathcal{O}(\dot{x}^3\varepsilon^3)) \\
      &= \sin(x) + \cos(x)\dot{x}\varepsilon
    \end{align*}
  \end{block}

  \begin{block}{$\cos$}
    \vspace{-0.5cm}
    \begin{align*}
      \cos(\xbf) &= \cos(x)\cos(\dot{x}\varepsilon) - \sin(x)\sin(\dot{x}\varepsilon) \\
      &= \cos(x)\left( 1 + \mathcal{O}(\dot{x}^2\varepsilon^2) \right) - \sin(x)(\dot{x}\varepsilon + \mathcal{O}(\dot{x}^3\varepsilon^3)) \\
      &= \cos(x) - \sin(x)\dot{x}\varepsilon
    \end{align*}
  \end{block}

  \begin{block}{$\sqrt{\cdot}$}
    \vspace{-0.5cm}
    \begin{align*}
      \sqrt{\xbf} &= \sqrt{x}\sqrt{1 + \frac{\dot{x}}{x}\varepsilon}
      &= \sqrt{x}\left( 1 + \frac{1}{2}\frac{\dot{x}}{x}\varepsilon + \mathcal{O}(\dot{x}^2\varepsilon^2) \right) \\
      &= \sqrt{x} + \frac{\dot{x}}{2\sqrt{x}}\varepsilon
    \end{align*}
  \end{block}
\end{frame}

\begin{frame}{Forward AD via Dual Numbers}

{\color{red}{Seed}} the input perturbation $\dif{x}/\dif{x} = 1$ in $\xbf = x + {\color{red}1}\cdot\varepsilon$, $\quad \varepsilon^2 = 0$.

Propagate through the program:
\begin{align*}
\ybf &= \cos(\xbf) = \cos(x) - \sin(x)\varepsilon,
&\dot{y} &= -\sin(x), \\
\zbf &= \sin(\xbf) = \sin(x) + \cos(x)\varepsilon,
&\dot{z} &= \cos(x), \\
\tbf_1 &= \ybf^2 = y^2 + 2y\dot{y}\varepsilon, 
&\dot{t_1} &= 2y\dot{y}, \\
\tbf_2 &= \zbf^2 = z^2 + 2z\dot{z}\varepsilon, 
&\dot{t_2} &= 2z\dot{z}, \\
\rbf &= \tbf_1 + \tbf_2 = r + \bigl(\dot{t_1} + \dot{t_2}\bigr)\varepsilon
&\dot{z} &= \cos(x), \dot{r} = \dot{t_1}+\dot{t_2}.
\end{align*}

The {\color{red} initialization} are called {\color{red} seeds}.

\end{frame}

\begin{frame}{Tangent, forward AD and directional derivatives}
  \begin{block}{Jacobian Vector products}
    Let $f: \mathbb{R}^p \to \mathbb{R}^d$ be a function implemented by program \texttt{P}. 
    The \alert{Jacobian vector product} (JVP) is a function mapping an input vector $\textbf{v}$ to the derivative of the function $f$ along the direction $\vbf$
    i.e.
    $$
    JVP: \mathbb{R}^p \times \mathbb{R}^p \to \mathbb{R}^d, JVP_f(\vbf; \wbf) = J_f(\wbf) \cdot \vbf
    $$
    No need for the evaluation or storage of the whole Jacobian.
  \end{block}
  
  \begin{block}{Remarks}
    This seems overly complicated however, note: 
    \begin{itemize}
      \item If $p = 1$, then usually one will have $\wbf$ represent the (scalar) data, and starting with $\vbf = 1$ will compute the whole \emph{Jacobian}.
      \item Assuming $d > 1$, one needs to know in which direction the propagation is required.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Why JVPs}
  \begin{block}{Newton--Krylov solvers}
    To solve a nonlinear system $F(\xbf)=0$, Newton's method uses
    $$
    J_F(\xbf_k)\,\delta_k = -F(\xbf_k), \qquad \xbf_{k+1} = \xbf_k + \delta_k.
    $$
    In large-scale problems, the linear system is typically solved by a Krylov method (e.g. GMRES).
  \end{block}

  \begin{itemize}
    \item For a linear system $J_F(\xbf_k)\,\delta_k = -F(\xbf_k)$, construct the Krylov subspace
      $$
      \mathcal{K}_m = \operatorname{span}\{r_0, J_F(\xbf_k)r_0, \ldots, J_F(\xbf_k)^{m-1}r_0\},
      \quad r_0 = F(\xbf_k).
      $$
    \item Equivalently, the method only requires evaluations of the form $J_F(\xbf_k)v$.
  \end{itemize}
\end{frame}

      \item Generate derivatives \emph{in vector mode}, allowing to mutualize expressions as much as possible.

\begin{frame}{Seed matrices}
  \begin{block}{Forward vector mode}
    Let $f:\mathbb{R}^p \to \mathbb{R}^d$ and let $S \in \mathbb{R}^{p\times r}$ be a \emph{seed matrix}.
    Initialize the tangent variables by
    $
    \dot{x} = S.
    $
    % or, equivalently, by setting the input perturbation to
    % $$
    % X(\varepsilon) = x + \sum_{j=1}^r \varepsilon_j s_j,
    % $$
    % where $s_j$ is the $j$-th column of $S$.

    Propagating these perturbations through the program gives
    $$
     \left[ \begin{array}{cccc} | & | & & | \\ \dot{\ybf}_1 & \dot{\ybf}_2 & \dots & \dot{\ybf}_r \\ | & | & & | \end{array} \right]\dot{\ybf} = J_f(\xbf)\,S \in \mathbb{R}^{q\times r}, \quad \dot{\ybf}_i = J_f(\xbf) \cdot \sbf_i.
    $$
    Thus each column of $\dot{y}$ is the derivative of the output in one chosen direction.

    \begin{itemize}
      \item If $r=1$, this is the usual JVP / directional derivative.
      \item If $S=I_p$, then $J_f(x)S = J_f(x)$ and the full Jacobian is obtained.
      \item If $r>1$, several directions are computed simultaneously, reducing overhead.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Left to right or right to left}

A program computes {\bf out} from {\bf in}:
\begin{equation*}
{\textbf{in}} \rightarrow {\tt v_1} \rightarrow {\tt v_2} \rightarrow \cdots \rightarrow {\tt v_9} \rightarrow {\textbf{out}}
\end{equation*}
One can propagate ${\dif{\tt v}\over \dif{\bf in}}$ forward $\Rightarrow$ Tangent (JVPs!):
\begin{equation*}
{\color{red}1.0} = \dot{\textbf{in}} = \frac{\dif{\textbf{in}}}{\dif{\textbf{in}}} \rightarrow \frac{\dif{\tt v_1}}{\dif{\textbf{in}}} \rightarrow\frac{\dif{\tt v_2}}{d\textbf{in}} \rightarrow \cdots \rightarrow \frac{\dif{\tt v_9}}{\dif\textbf{in}} \rightarrow \frac{\dif{\textbf{out}}}{\dif{\textbf{in}}}
\end{equation*}
One can propagate ${\dif{\bf out}\over \dif{\tt v}}$ backward $\Rightarrow$ Adjoint (VJPs, coming now):
\begin{equation*}
\frac{{\dif{\textbf{out}}}}{\dif{\textbf{in}}} \leftarrow \frac{\dif{\textbf{out}}}{\dif{\tt v_1}} \leftarrow \frac{\dif{\textbf{out}}}{\dif{\tt v_2}} \leftarrow \cdots \leftarrow \frac{{\dif\textbf{out}}}{\dif{\tt v_9}} \leftarrow \frac{\dif{\textbf{out}}}{\dif{\textbf{out}}} = \bar{\textbf{out}} = {\color{red}1.0}
\end{equation*}
Note that the initialization might not be just a scalar $1$, but could be any seed matrix when dealing with many variables!
\end{frame}

\begin{frame}{JVPs vs VJPs}
  \begin{block}{JVPs as directional derivatives}
    $$JVP_f(\wbf, \vbf) = J_F(\wbf)\cdot \vbf = \lim_{h \to 0} \frac{f(\wbf + h \vbf) - f(\wbf)}{h}$$ 
    It describes the changes in the outputs given the perturbation of the inputs $\vbf$.
  \end{block}
  \begin{block}{VJPs as differentiated projections}
    Consider $\ubf \in \Rbb^{d}$ and let $g(\wbf; \ubf) = \langle u, f(\wbf) \rangle$ be the projection of the outputs of $f$ onto the (output) direction $\ubf$. Its gradient is given by 
    \begin{align*}
      \nabla_\wbf g(\wbf; \ubf) = \sum_{i=1}^d u_i \nabla_\wbf f_i(\wbf) = \ubf^T J_f(\wbf).
    \end{align*}
  \end{block}
  JVPs are precisely the projection of the perturbations in the inputs onto the objective values, while JVPs are precisely the propagation of output sensitivities to input variables.
\end{frame}

\begin{frame}{VJPs... reverse... adjoints}
  \begin{block}{reverse}
    From the definition of the JVPs we can see that \alert{function values perturbations} propagate to input pertubations. 
    This is also called \alert{reverse} or \alert{backward} propagation (a.k.a. backprograpation).
  \end{block}
  \begin{block}{adjoint}
    Output sensitivities are usually called \alert{adjoint variables} and denoted with bar: $\bar{\cdot}$. 
    The name adjoint comes from the scalar product relationship, for any $\dot{\wbf} \in \mathbb{R}^{n}$ and $\bar{\wbf} \in mathbb{R}^d$,
    $$
    \langle J_f(\wbf) \dot{\wbf}, \bar{\wbf} \rangle = \langle \dot{\wbf}, J_f(\wbf)^T\bar{\wbf} \rangle
    $$
    i.e. JVP's adjoint is VJP.
  \end{block}
\end{frame}

\begin{frame}{JVPs complement VJPs}
  \begin{block}{A quadratic model and conjugate gradient}
    For a quadratic objective
    $$
    \phi(\xbf) = \tfrac12 \xbf^T A \xbf - b^T\xbf,
    $$
    the conjugate gradient method generates iterates
    $$
    \xbf_{k+1} = \xbf_k + \alpha_k p_k
    $$
    where the search direction satisfies
    $$
    p_{k+1} = r_{k+1} + \beta_k p_k,
    \qquad
    r_k = b - A\xbf_k.
    $$
  \end{block}

  \begin{itemize}
    \item In a nonlinear setting, one often replaces $A$ by the Hessian $\nabla^2 \phi(\xbf_k)$.
    \item The Hessian-vector product needed by CG is then of the form
      $$
      \nabla^2 \phi(\xbf_k) v.
      $$
    \item For a scalar residual model $r(\xbf)$, this can be obtained from JVPs and VJPs via
      $$
      \nabla^2 \phi(\xbf) v = J_r(\xbf)^T J_r(\xbf) v,
      $$
      so the method needs repeated Jacobian action on vectors.
    \item This is exactly why JVPs and VJPs are useful inside Newton--Krylov and CG-type solvers.
  \end{itemize}
\end{frame}

\begin{frame}{General Reverse Rule}

For any variable $v$:

\[
\bar{v} = \sum_{w\in children(v)}
\bar{w}\frac{\partial w}{\partial v}
\]

\begin{block}{Interpretation}
Reverse mode sums contributions from all downstream paths.
\end{block}

\end{frame}

\begin{block}{Rule of thumb}
Cost $\propto$ number of outputs
\end{block}

Best when:

\begin{itemize}
\item Many inputs
\item Scalar (or few) output(s)
\end{itemize}
